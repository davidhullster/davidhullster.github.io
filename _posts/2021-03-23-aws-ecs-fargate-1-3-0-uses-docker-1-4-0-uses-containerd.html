In terraform the ECS version can be specified with 'platform_version'

resource "aws_ecs_service" "main" {
name = "tools-ecs-service" cluster = aws_ecs_cluster.main.id
task_definition = aws_ecs_task_definition.app.arn
desired_count = var.app_count
launch_type = "FARGATE"
platform_version = "1.3.0"
