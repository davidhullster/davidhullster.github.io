/home/<USER>/snap/spotify/current/.config/spotify/prefs

Set app.window.position.saved value to fix maximized window at startup

app.window.position.saved=false

>> :~/snap/spotify$ cat current/.config/spotify/prefs
autologin.blob=<long_string>
autologin.saved_credentials="{\"<long_string2>\"]}"
autologin.canonical_username="<long_string3>"
app.window.position.x=72
autologin.username=""<long_string3>
app.window.position.saved=true
app.window.position.height=836
app.autostart-mode=""
app.window.position.y=64
storage.last-location="/home/<user_name>/snap/spotify/common/.cache/spotify/Storage"
app.window.position.width=1528
core.clock_delta=1
app.last-launched-version="1.1.46.916.g416cacf1"
app.autostart-configured=true
