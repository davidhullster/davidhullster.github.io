!/bin/bash -xe
exec > >(tee /var/log/user-data.log|logger -t user-data -s 2>/dev/console) 2>&1
yum -y update
echo "#### USERDATA START ####"
sudo yum install httpd -y
sudo systemctl start httpd.service
sudo systemctl enable httpd.service
